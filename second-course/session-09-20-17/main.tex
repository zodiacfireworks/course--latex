\newcommand{\fibonacci}[1]{
	\counter=#1
	\fone=1
	\ftwo=1
	\temp=0
	\the\fone, \the\ftwo
	\fibloop
}

\newcommand{\fibloop}{
	\let\next=\fibloop
	\temp=\fone
	\fone=\ftwo
	\advance \ftwo by \temp
	\ifnum\counter
		\let\next=\relax
	\else
		\advance \counter by -1
	\fi
	\the\two
	\next
}
